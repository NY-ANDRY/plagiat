    <script>
        require.config({ paths: { vs: 'https://unpkg.com/monaco-editor/min/vs' } });
        let editor;

        require(['vs/editor/editor.main'], function () {
            editor = monaco.editor.create(document.getElementById('editor'), {
                value: '// Sélectionnez un fichier dans l\'explorateur pour afficher son contenu.',
                language: 'javascript',
                theme: 'vs-dark',
                readOnly: true,
                automaticLayout: true,
                fontFamily: "'JetBrains Mono', 'Fira Code', Consolas, monospace",
                fontSize: 14,
                minimap: { enabled: false },
                scrollBeyondLastLine: false,
                renderLineHighlight: "all"
            });

            fetch('/zip/tree/projet.zip')
                .then(res => res.json())
                .then(files => {
                    const tree = buildTree(files);
                    const container = document.getElementById('file-tree');
                    container.innerHTML = '';
                    
                    renderTree(tree, container, 0);
                    lucide.createIcons();
                })
                .catch(err => {
                    document.getElementById('file-tree').innerHTML = '<div class="p-4 text-red-500 text-sm">Erreur de chargement</div>';
                });
        });

        function buildTree(files) {
            const root = [];
            files.forEach(path => {
                let parts = path.split('/');
                let currentLevel = root;
                
                for (let i = 0; i < parts.length; i++) {
                    let part = parts[i];
                    let existingPath = currentLevel.find(p => p.name === part);
                    
                    if (existingPath) {
                        currentLevel = existingPath.children;
                    } else {
                        let isFile = (i === parts.length - 1);
                        let newPart = { 
                            name: part, 
                            isFile: isFile, 
                            path: isFile ? path : null, 
                            children: [] 
                        };
                        currentLevel.push(newPart);
                        currentLevel = newPart.children;
                    }
                }
            });
            return root;
        }

        function getFileIcon(name) {
            const lower = name.toLowerCase();
            if (lower.endsWith('.json')) return 'file-json';
            if (lower.match(/\.(php|js|css|html|blade\.php|ts|vue|jsx)$/)) return 'file-code';
            if (lower.match(/\.(svg|png|jpg|jpeg|gif|webp)$/)) return 'image';
            if (lower.match(/\.(md|txt)$/)) return 'file-text';
            return 'file';
        }
        
        function getIconColor(name) {
            const lower = name.toLowerCase();
            if (lower.endsWith('.php')) return 'text-indigo-400';
            if (lower.endsWith('.js')) return 'text-yellow-400';
            if (lower.endsWith('.json')) return 'text-yellow-200';
            if (lower.endsWith('.css')) return 'text-blue-400';
            if (lower.endsWith('.html') || lower.endsWith('.blade.php')) return 'text-orange-400';
            if (lower.match(/\.(svg|png|jpg|jpeg)$/)) return 'text-emerald-400';
            return 'text-gray-400';
        }

        function renderTree(nodes, container, depth) {
            nodes.sort((a, b) => {
                if (a.isFile !== b.isFile) return a.isFile ? 1 : -1;
                return a.name.localeCompare(b.name);
            });

            nodes.forEach(node => {
                const el = document.createElement('div');
                const header = document.createElement('div');
                header.className = 'file-item select-none';
                header.style.paddingLeft = (depth * 12 + 10) + 'px';

                const defaultExpanded = depth < 2;
                
                if (!node.isFile) {
                    const chIcon = document.createElement('i');
                    chIcon.setAttribute('data-lucide', defaultExpanded ? 'chevron-down' : 'chevron-right');
                    chIcon.className = 'chevron';
                    header.appendChild(chIcon);
                    
                    const fIcon = document.createElement('i');
                    fIcon.setAttribute('data-lucide', defaultExpanded ? 'folder-open' : 'folder');
                    fIcon.className = 'icon text-blue-400';
                    header.appendChild(fIcon);
                } else {
                    const spacer = document.createElement('div');
                    spacer.className = 'w-[14px] flex-shrink-0';
                    header.appendChild(spacer);
                    
                    const fIcon = document.createElement('i');
                    fIcon.setAttribute('data-lucide', getFileIcon(node.name));
                    fIcon.className = 'icon ' + getIconColor(node.name);
                    header.appendChild(fIcon);
                }

                const text = document.createElement('span');
                text.className = 'truncate';
                text.innerText = node.name;
                header.appendChild(text);

                el.appendChild(header);

                if (!node.isFile) {
                    const childrenContainer = document.createElement('div');
                    childrenContainer.className = 'tree-children' + (defaultExpanded ? ' open' : '');
                    el.appendChild(childrenContainer);
                    
                    renderTree(node.children, childrenContainer, depth + 1);

                    header.addEventListener('click', (e) => {
                        e.stopPropagation();
                        const isOpen = childrenContainer.classList.contains('open');
                        
                        if (isOpen) {
                            childrenContainer.classList.remove('open');
                            header.children[0].setAttribute('data-lucide', 'chevron-right');
                            header.children[1].setAttribute('data-lucide', 'folder');
                        } else {
                            childrenContainer.classList.add('open');
                            header.children[0].setAttribute('data-lucide', 'chevron-down');
                            header.children[1].setAttribute('data-lucide', 'folder-open');
                        }
                        lucide.createIcons({
                            nameAttr: 'data-lucide',
                            root: header
                        });
                    });
                } else {
                    header.addEventListener('click', (e) => {
                        e.stopPropagation();
                        
                        document.querySelectorAll('.file-item.active').forEach(elem => {
                            elem.classList.remove('active');
                        });
                        header.classList.add('active');
                        
                        loadFile(node.path, node.name);
                    });
                }

                container.appendChild(el);
            });
        }

        function getLanguage(filename) {
            const ext = filename.split('.').pop().toLowerCase();
            const map = {
                'php': 'php', 'js': 'javascript', 'ts': 'typescript', 
                'json': 'json', 'css': 'css', 'html': 'html', 'md': 'markdown',
                'xml': 'xml', 'sql': 'sql', 'yaml': 'yaml', 'yml': 'yaml'
            };
            return map[ext] || 'plaintext';
        }

        function loadFile(path, filename) {
            const tab = document.getElementById('editor-tab');
            const fname = document.getElementById('editor-filename');
            tab.classList.remove('hidden');
            tab.classList.add('flex');
            tab.style.borderTopColor = '#007acc';
            fname.innerText = filename;
            
            const tabIcon = tab.querySelector('i');
            tabIcon.setAttribute('data-lucide', getFileIcon(filename));
            tabIcon.className = 'w-4 h-4 ' + getIconColor(filename);
            lucide.createIcons({ root: tab });
            
            editor.setValue('// Chargement du fichier...');

            fetch('/zip/file/projet.zip?path=' + path)
                .then(res => res.text())
                .then(code => {
                    const lang = getLanguage(filename);
                    monaco.editor.setModelLanguage(editor.getModel(), lang);
                    editor.setValue(code);
                })
                .catch(() => {
                    editor.setValue('// Impossible de charger ce fichier.');
                });
        }
